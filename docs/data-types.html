2.2 Data types

Atomic

In most cases, each atomic element has a type (mode) of;

  • numeric: number

  • logical: TRUE or FALSE (T or F for shortcuts)

  • character: character string

  • factor: a level of categorical variable

Other types include date and nonexistent NULL. Factor is also a class of its own, meaning that many R functions apply special operations to the objects under the factor class.

c(is.numeric(123), is.numeric("abc"), is.numeric(TRUE))
## [1]  TRUE FALSE FALSE
c(is.logical(123), is.logical("abc"), is.logical(TRUE))
## [1] FALSE FALSE  TRUE
c(is.character(123), is.character("abc"), is.character(TRUE))
## [1] FALSE  TRUE FALSE
# "<-" means an assignment from right to left
# () brackets at the most outer layer means to print 
( factor1 <- as.factor(c(1,2,3)) )  # Looks like numeric but not
## [1] 1 2 3
## Levels: 1 2 3
( factor2 <- as.factor(c("a","b","c")) )  # Looks like characters but not
## [1] a b c
## Levels: a b c
( factor3 <- as.factor(c(TRUE,FALSE,T)) ) # Looks like logicals but not
## [1] TRUE  FALSE TRUE 
## Levels: FALSE TRUE
# Subset the first element of factor1 as factor1[1]  etc. 
c(is.numeric(factor1[1]), is.logical(factor1[1]), is.character(factor1[1]))
## [1] FALSE FALSE FALSE
c(is.numeric(factor2[1]), is.logical(factor2[1]), is.character(factor2[1]))
## [1] FALSE FALSE FALSE
c(is.numeric(factor3[1]), is.logical(factor3[1]), is.character(factor3[1]))
## [1] FALSE FALSE FALSE
c(is.factor(factor1[1]), is.factor(factor2[1]), is.factor(factor3[1]))
## [1] TRUE TRUE TRUE
# There are length-zero elements for numeric, logical, and character
length(NULL) 
## [1] 0
length(numeric(0))
## [1] 0
length(logical(0))
## [1] 0
length(character(0))
## [1] 0
length(factor(0)) # has length 1
## [1] 1

Each vector has a type of numeric, logical, character, or factor. Each matrix has a type of numeric, logical, or character. A data frame can contain mixed types in columns where each column (e.g., a variable) has a type of numeric, logical, character or factor.

( vector1 <- c(1, NA, 2, 3) ) # read as numeric
## [1]  1 NA  2  3
( vector2 <- c(TRUE, FALSE, T, F) )# read as logical
## [1]  TRUE FALSE  TRUE FALSE
( vector3 <- c(1, NA, "abc", TRUE, "TRUE") ) # read as character
## [1] "1"    NA     "abc"  "TRUE" "TRUE"
( vector4 <- as.factor(c(1, NA, "abc", TRUE, "TRUE")) ) # read as factor 
## [1] 1    <NA> abc  TRUE TRUE
## Levels: 1 abc TRUE
( matrix1 <- matrix(c(1:6), nrow = 3) )  # read as character
##      [,1] [,2]
## [1,]    1    4
## [2,]    2    5
## [3,]    3    6
( matrix2 <- matrix(c(TRUE,FALSE,rep(T,3),F), nrow = 3) )  # read as logical
##       [,1]  [,2]
## [1,]  TRUE  TRUE
## [2,] FALSE  TRUE
## [3,]  TRUE FALSE
( matrix3 <- matrix(c(1,2,3,"a","b","abc"), nrow = 3) )  # read as character
##      [,1] [,2] 
## [1,] "1"  "a"  
## [2,] "2"  "b"  
## [3,] "3"  "abc"
( df1 <- data.frame(
                num  = c(1,2,3),           # read as numeric

Factor

A factor object can be ordered (use ordered()) or unordered (use factor()) . The difference is that certain statistical analyses and graphical procedures treat orderd factor objects in the specified order.

df <- data.frame(
      fac1 = c(0,1,1,4,4,2,2,3),
      fac2 = c(1,2,3,1,1,2,2,3),
      fac3 = c(4,2,3,4,4,2,2,3)
      )

# order is given by the listing order in levels 
df$fac1 <- ordered(df$fac1,
                  levels = c(0,4,3,2,1) 
                  ) 

# attach labels to factors: 1=red, 2=blue, 3=green
# in factor(), there is no particular order
df$fac2 <- factor(df$fac2,
                  levels = c(1,2,3),
                  labels = c("red", "blue", "green")
                  ) 
                  
# attach labels to factors: 2=Low, 3=Medium, 4=High
df$fac3 <- ordered(df$fac3,
                  levels = c(2,3,4),
                  labels = c("Low", "Medium", "High")
                  ) 

df$fac1
## [1] 0 1 1 4 4 2 2 3
## Levels: 0 < 4 < 3 < 2 < 1
summary(df$fac1)  # gives the table of counts for each level
## 0 4 3 2 1 
## 1 2 1 2 2
df$fac2
## [1] red   blue  green red   red   blue  blue  green
## Levels: red blue green
summary(df$fac2)
##   red  blue green 
##     3     3     2
df$fac3
## [1] High   Low    Medium High   High   Low    Low    Medium
## Levels: Low < Medium < High
summary(df$fac3)
##    Low Medium   High 
##      3      2      3

Matrix

matrix() defines a matrix from a vector. The default is to arrange the vector by column (byrow = FALSE option).

# the default byrow = FALSE 
matrix(data = c(1:6), nrow = 2, ncol = 3, byrow = FALSE, dimnames = NULL)
##      [,1] [,2] [,3]
## [1,]    1    3    5
## [2,]    2    4    6
# the default byrow = TRUE 
matrix(data = c(1:6), nrow = 2, ncol = 3, byrow = TRUE, dimnames = NULL)
##      [,1] [,2] [,3]
## [1,]    1    2    3
## [2,]    4    5    6
# give row and column names to a matrix 
mat1 <- matrix(data = c(1:6), nrow = 2, ncol = 3, byrow = FALSE, 
       dimnames = list(c("r1","r2"), c("c1","c2","c3")))
mat1
##    c1 c2 c3
## r1  1  3  5
## r2  2  4  6
dim(mat1)  # dimention: row by column
## [1] 2 3
colnames(mat1)
## [1] "c1" "c2" "c3"
rownames(mat1)
## [1] "r1" "r2"
colnames(mat1) <- c("v1","v2","v3")  # change column names by assignment "<-" 
mat1
##    v1 v2 v3
## r1  1  3  5
## r2  2  4  6
# R will make a guess when only nrow or ncol is supplied
matrix(data = c(1:6), nrow = 2)
##      [,1] [,2] [,3]
## [1,]    1    3    5
## [2,]    2    4    6
matrix(data = c(1:6), ncol = 3)
##      [,1] [,2] [,3]
## [1,]    1    3    5
## [2,]    2    4    6
# combine matrix by column "cbind()" or by row "rbind()"
cbind(mat1,mat1)
##    v1 v2 v3 v1 v2 v3
## r1  1  3  5  1  3  5
## r2  2  4  6  2  4  6
rbind(mat1,mat1)
##    v1 v2 v3
## r1  1  3  5
## r2  2  4  6
## r1  1  3  5
## r2  2  4  6

##    v1 v2 v3
## r1  0  7  5
## r2  0  8  6

## v1 v2 v3 
##  0  7  5

## r1 r2 
##  0  0

## v1 v2 v3 
##  0  7  5

## r1  0  5
## r2  0  6

## [1] 6
# Important: 
# When a single row or column is extracted,

apply() function takes a matrix and applies a function for a specified margin (dimention index number).

mat1
##    v1 v2 v3
## r1  0  7  5
## r2  0  8  6
apply(mat1,1,mean)  # dimention 1 (across rows)
##       r1       r2 
## 4.000000 4.666667
apply(mat1,2,mean)  # dimention 2 (across columns)
##  v1  v2  v3 
## 0.0 7.5 5.5
# one can write a custom function inside apply(). (called annonymous function)    
# Its argument corresponds to the row or column vector passed by apply(). 
apply(mat1,2, function(x) sum(x)/length(x) )  # x is the internal name
##  v1  v2  v3 
## 0.0 7.5 5.5
ans1 <- apply(mat1,2, function(x) {   # a longer custom function with { }
                         avg = mean(x)
                         sd = sd(x)
                         z = (x - mean(x))/sd(x)  
                          # return the results as a list
                         list(avg = avg, sd = sd, z = z)
                      }
        )

unlist(ans1[[2]])  # results for the second column  
##        avg         sd       z.r1       z.r2 
##  7.5000000  0.7071068 -0.7071068  0.7071068
unlist(ans1[[3]])  # results for the third column  
##        avg         sd       z.r1       z.r2 
##  5.5000000  0.7071068 -0.7071068  0.7071068

Data Frame

A data frame is similar to a matrix, but it accepts multiple types (modes) of variables across columns as in the case of typical data analysis programs like SAS, SPSS, Stata etc. In some cases, matrices and data frames can be treated interchangeably. In other cases, we have to pay attention to which class an object belongs to. Many data manipulation funtions are written for data frames, while some base R functions are written for matrices.

( mymat1 <- matrix(data = c(1:6), nrow = 2, ncol = 3, 
       dimnames = list(c("r1","r2"), c("c1","c2","c3"))) )
##    c1 c2 c3
## r1  1  3  5
## r2  2  4  6
class(mymat1)
## [1] "matrix"
colnames(mymat1)
## [1] "c1" "c2" "c3"
names(mymat1)
## NULL
( mydf1 <- data.frame(
                mymat1, 
                num  = c(1,2),         
                fac1 = c("a","abc"),    
                logi = c(TRUE, FALSE), 
                fac2  = c(1,"a")   
              )  
)
##    c1 c2 c3 num fac1  logi fac2
## r1  1  3  5   1    a  TRUE    1
## r2  2  4  6   2  abc FALSE    a
class(mydf1)
## [1] "data.frame"
colnames(mydf1)
## [1] "c1"   "c2"   "c3"   "num"  "fac1" "logi" "fac2"
names(mydf1)   # colnames and names are the same 
## [1] "c1"   "c2"   "c3"   "num"  "fac1" "logi" "fac2"

Subsetting a data frame is similar but there are a few additional methods.

mydf1[1,]   # row = 1 and all columns 
##    c1 c2 c3 num fac1 logi fac2
## r1  1  3  5   1    a TRUE    1
mydf1[,1]   # all rows and col = 1
## [1] 1 2
# data frame preserves dimension while subsetting a row but not a column
dim(mydf1[1,])  
## [1] 1 7
dim(mydf1[,1])  
## NULL
dim(mydf1[,1,drop=FALSE])   # use drop = FALSE to keep a column vector
## [1] 2 1
mydf1[,1,drop=FALSE]
##    c1
## r1  1
## r2  2
mydf1[, c('c1','num','logi')]
##    c1 num  logi
## r1  1   1  TRUE
## r2  2   2 FALSE
class(mydf1[, c('c1','num','logi')])
## [1] "data.frame"
# extraction by column name with "$" symbole:  df$varname  
mydf1$c1
## [1] 1 2
dim(mydf1$c1)
## NULL
# one can use quote ' ' or " " as well 
mydf1$'c1'
## [1] 1 2
# similarly, extraction by column name with [[ ]]: df[['varname']]    
mydf1[['c1']]
## [1] 1 2
dim(mydf1[['c1']])
## NULL
# or by index 
mydf1[[1]]
## [1] 1 2
# [[ ]] method is useful when passing a variable name as a string

set_to_na <- function(df, var) {
                df[[var]] <- NA  
                df  # the last expression inside the function is returned
              }

mydf1
##    c1 c2 c3 num fac1  logi fac2
## r1  1  3  5   1    a  TRUE    1
## r2  2  4  6   2  abc FALSE    a

mydf2
##    c1 c2 c3 num fac1  logi fac2
## r1  1 NA  5   1    a  TRUE    1
## r2  2 NA  6   2  abc FALSE    a

mydf1
##    c1 c2 c3 num fac1  logi fac2 newvar
## r1  1  3  5   1    a  TRUE    1      4
## r2  2  4  6   2  abc FALSE    a      4

mydf1
##    c1 c2 c3 num fac1  logi fac2 newvar newvar2
## r1  1  3  5   1    a  TRUE    1      4       8
## r2  2  4  6   2  abc FALSE    a      4      10

apply() may work with data frames, but data frames are not exactly matrices. Use simplified apply sapply() or list apply lapply() that tends to work better for data frames in many cases.

mydf1 
##    c1 c2 c3 num fac1  logi fac2 newvar newvar2
## r1  1  3  5   1    a  TRUE    1      4       8
## r2  2  4  6   2  abc FALSE    a      4      10
# unlike apply(), lapply() does not use a margin (dimension) argument 
( idx_num2 <- unlist(lapply(mydf1, is.numeric)) )
##      c1      c2      c3     num    fac1    logi    fac2  newvar newvar2 
##    TRUE    TRUE    TRUE    TRUE   FALSE   FALSE   FALSE    TRUE    TRUE
# the same result from sapply(): a user-friendly (simplified) version of lapply 
# sapply() makes a guess on the margin (dimension) to apply the function
( idx_num <- sapply(mydf1, is.numeric) )
##      c1      c2      c3     num    fac1    logi    fac2  newvar newvar2 
##    TRUE    TRUE    TRUE    TRUE   FALSE   FALSE   FALSE    TRUE    TRUE
apply(mydf1[,idx_num], 2, mean)
##      c1      c2      c3     num  newvar newvar2 
##     1.5     3.5     5.5     1.5     4.0     9.0
sapply(mydf1[,idx_num], mean)
##      c1      c2      c3     num  newvar newvar2 
##     1.5     3.5     5.5     1.5     4.0     9.0
unlist(lapply(mydf1[,idx_num], mean))
##      c1      c2      c3     num  newvar newvar2 
##     1.5     3.5     5.5     1.5     4.0     9.0

List

A list is an ordered collection of (possibly unrelated) objects separated by comma. The objects in a list are referenced by [[1]], [[2]], …, or [[‘var1’]], [[‘var2’]], … etc.

mylist1 <- list(v1 = c(1,2,3),
             v2 = c("a","b"),
             v3 = factor(c("blue","red","orange","yellow")),
             v4 = data.frame( u1 = c(1:3), u2 = c("p","q","r"))
             )

lapply(mylist1, function(x) {

                  if (is.character(x)) return(x)
