Visualizing Model Fit

Raw data

# raw data plot 
base_plot <- 
  ChickWeight2 %>%
    ggplot(aes( x = Time, 
                y = weight, 
                color = Diet)) + 
    geom_jitter(size =.75, alpha=.5) 

base_plot

Model 1 fit (approx.)

# overlay model_1, approximated by  y ~ x
base_plot +
  geom_smooth(
    aes(x = Time, 
        y = fitted.values(model_1),
        color = Diet),
    method='lm', formula = y ~ x, se=FALSE)

Model 2 fit (approx.)

# overlay model_2, approximated by loess 
base_plot +
  geom_smooth(
    aes(x = Time, 
        y = fitted.values(model_2),
